== Приклади коду для редагування

[source, html]
----
<!DOCTYPE html>
<html>
<head>
    <link rel="stylesheet" href="css/style.css">
</head>
<body>
    <div class="center">
        <img src="images/ua.png" class="trinity"/>
    </div>
    <p class="header text14">
        ...
    </p>
    <table>
        <colgroup>
            <col style="width: 40%;">
            <col style="width: 60%;">
        </colgroup>
        <tbody>
            <tr>
                <td>...</td>
                <td class="value">...</td>
            </tr>
        </tbody>
    </table>
</body>
</html>
----
